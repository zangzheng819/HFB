import numpy as np

import matplotlib.pyplot as plt


def solve_harper_equation(q, t, phi):

    size = q  
    H = np.zeros((size, size), dtype=complex)

    for j in range(size):

        H[j, (j + 1) % size] = -t

        H[(j + 1) % size, j] = -t

        H[j, j] = 2*t*np.cos(2 * np.pi * phi * j + np.pi / 2)



    
    eigenvalues = np.linalg.eigvalsh(H)

    return eigenvalues



def plot_hofstadter_butterfly(q, t, max_phi, num_points):

    phis = np.linspace(0, max_phi, num_points)

    energies = []

    for phi in phis:

        eigvals = solve_harper_equation(q, t, phi)

        energies.append(eigvals)



    energies = np.array(energies).T 



 
    plt.figure(figsize=(8, 6))

    for eig in energies:

        plt.plot(phis, eig, 'k.', markersize=1)



    plt.title('Hofstadter Butterfly')

    plt.xlabel('$\phi$')

    plt.ylabel('energy')

    plt.grid(True)

    plt.show()



if __name__ == '__main__':

    q = 100  

    t = 1  
    plot_hofstadter_butterfly(q, t, 1, 500)
